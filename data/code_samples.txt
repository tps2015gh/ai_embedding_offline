func CreateEmbeddings(texts []string, dimensions int) ([]Vector, error) {
	vectors := make([]Vector, 0, len(texts))
	for _, text := range texts {
		embedding := generateEmbedding(text, dimensions)
		vectors = append(vectors, Vector{
			Text:      text,
			Embedding: embedding,
		})
	}
	return vectors, nil
}

func generateEmbedding(text string, dimensions int) []float64 {
	embedding := make([]float64, dimensions)
	for i := 0; i < dimensions; i++ {
		embedding[i] = float64(i) / float64(dimensions)
	}
	return embedding
}

func CosineSimilarity(a, b []float64) float64 {
	dotProduct := 0.0
	normA := 0.0
	normB := 0.0
	for i := range a {
		dotProduct += a[i] * b[i]
		normA += a[i] * a[i]
		normB += b[i] * b[i]
	}
	return dotProduct / (normA * normB)
}

func SearchSimilar(queryText string, limit int) ([]VectorRecord, error) {
	results, err := vectorstore.SearchSimilar(queryText, limit)
	if err != nil {
		return nil, err
	}
	return results, nil
}

func handleSearch(w http.ResponseWriter, r *http.Request) {
	var req struct {
		Query string `json:"query"`
		Limit int    `json:"limit"`
	}
	if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
		sendError(w, "Invalid request", http.StatusBadRequest)
		return
	}
	results, err := vectorstore.SearchSimilar(req.Query, req.Limit)
	if err != nil {
		sendError(w, err.Error(), http.StatusInternalServerError)
		return
	}
	sendResponse(w, results)
}

func handleNGramPredict(w http.ResponseWriter, r *http.Request) {

}
